41  /**
42   * This is used to test IOUtil for correctness. The following checks are performed:
43   * <ul>
44   * <li>The return must not be null, must be the same type and equals() to the method's second arg</li>
45   * <li>All bytes must have been read from the source (available() == 0)</li>
46   * <li>The source and destination content must be identical (byte-wise comparison check)</li>
47   * <li>The output stream must not have been closed (a byte/char is written to test this, and subsequent size
48   * checked)</li>
49   * </ul>
50   * Due to interdependencies in IOUtils and IOUtilsTestlet, one bug may cause multiple tests to fail.
51   *
52   * @author <a href="mailto:jefft@apache.org">Jeff Turner</a>
53   */

95      private void createFile( File file, long size )
96          throws IOException
97      {
98          BufferedOutputStream output = new BufferedOutputStream( new FileOutputStream( file ) );
99  
100         for ( int i = 0; i < size; i++ )
101         {
102             output.write( (byte) ( i % 128 ) ); // nice varied byte pattern compatible with Readers and Writers
103         }
104 
105         output.close();
106     }

111         assertTrue( "Content not equal according to java.util.Arrays#equals()", Arrays.equals( b0, b1 ) );
112     }

114     /** Assert that the content of two files is the same. */
115     private void assertEqualContent( File f0, File f1 )
116         throws IOException
117     {
118         FileInputStream is0 = new FileInputStream( f0 );
119         FileInputStream is1 = new FileInputStream( f1 );
120         byte[] buf0 = new byte[FILE_SIZE];
121         byte[] buf1 = new byte[FILE_SIZE];
122         int n0 = 0;
123         int n1 = 0;
124 
125         try
126         {
127             while ( 0 <= n0 )
128             {
129                 n0 = is0.read( buf0 );
130                 n1 = is1.read( buf1 );
131                 assertTrue( "The files " + f0 + " and " + f1 + " have differing number of bytes available (" + n0
132                     + " vs " + n1 + ")", ( n0 == n1 ) );
133 
134                 assertTrue( "The files " + f0 + " and " + f1 + " have different content", Arrays.equals( buf0, buf1 ) );
135             }
136         }
137         finally
138         {
139             is0.close();
140             is1.close();
141         }
142     }

144     /** Assert that the content of a file is equal to that in a byte[]. */
145     private void assertEqualContent( byte[] b0, File file )
146         throws IOException
147     {
148         FileInputStream is = new FileInputStream( file );
149         byte[] b1 = new byte[b0.length];
150         int numRead = is.read( b1 );
151         assertTrue( "Different number of bytes", numRead == b0.length && is.available() == 0 );
152         for ( int i = 0; i < numRead; assertTrue( "Byte " + i + " differs (" + b0[i] + " != " + b1[i] + ")",
153                                                   b0[i] == b1[i] ), i++ )
154             ;
155         is.close();
156     }

163         FileInputStream fin = new FileInputStream( testFile );
164         FileOutputStream fout = new FileOutputStream( destination );
165

169

516     private File newFile( String filename )
517         throws Exception
518     {
519         File destination = new File( testDirectory, filename );
520         assertTrue( filename + "Test output data file shouldn't previously exist", !destination.exists() );
521 
522         return destination;

525     private void checkFile( File file )
526         throws Exception
527     {
528         assertTrue( "Check existence of output file", file.exists() );
529         assertEqualContent( testFile, file );
530     }
531 
532     private void checkWrite( OutputStream output )
533         throws Exception
534     {
535         try
536         {
537             new PrintStream( output ).write( 0 );
538         }
539         catch ( Throwable t )
540         {
541             throw new Exception( "The copy() method closed the stream " + "when it shouldn't have. "
542                 + t.getMessage() );
543         }
544     }
545 
546     private void checkWrite( Writer output )
547         throws Exception
548     {
549         try
550         {
551             new PrintWriter( output ).write( 'a' );
552         }
553         catch ( Throwable t )
554         {
555             throw new Exception( "The copy() method closed the stream " + "when it shouldn't have. "
556                 + t.getMessage() );
557         }
558     }

560     private void deleteFile( File file )
561         throws Exception
562     {
563         assertTrue( "Wrong output size: file.length()=" + file.length() + "!=" + FILE_SIZE + 1,
564                     file.length() == FILE_SIZE + 1 );
565 
566         assertTrue( "File would not delete", ( file.delete() || ( !file.exists() ) ) );
567     }
